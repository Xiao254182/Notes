2.使用脚本对五台主机做初始化(执行此脚本前需保证五台主机的ip地址已做绑定)

```shell
#!/bin/bash

# 安装 sshpass
yum install -y sshpass

# 修改 ansible 的 hosts 文件
cd /etc/ansible
cat > hosts << EOF
[host]
192.168.100.11 #proxy-master
192.168.100.12 #proxy-slave
192.168.100.13 #master
192.168.100.14 #worker
192.168.100.41 #ops
EOF

# 从 hosts 文件中读取主机列表和主机名
hosts=($(awk '{print $1}' /etc/ansible/hosts | sed 's/\[host\]//g' | grep .))
hostnames=($(awk '{print $2}' /etc/ansible/hosts | sed 's/#//g' | sed 's/ //g' | grep .))

# 生成 SSH 密钥对
ssh-keygen -t rsa -b 2048 -N "" -f ~/.ssh/id_rsa

# 配置 SSH 免密码登录并设置主机名
for ((i=0; i<${#hosts[@]}; i++)); do
  host=${hosts[$i]}
  hostname=${hostnames[$i]}
  
  #将当前主机的公钥复制到其他主机
  sshpass -p '000000' ssh-copy-id -o StrictHostKeyChecking=no root@${hosts[$i]}
  # 设置主机名和配置系统
  ansible -i /etc/ansible/hosts $host -u root -m shell -a "hostnamectl set-hostname $hostname && systemctl stop firewalld && systemctl disable firewalld && sed -i 's/^SELINUX=.*/SELINUX=disabled/g' /etc/selinux/config && setenforce 0 && swapoff -a && timedatectl set-timezone Asia/Shanghai && timedatectl set-ntp true && yum install -y epel-* && yum install -y vim wget net-tools telnet tree nmap sysstat lrzsz dos2unix bind-utils"
done

echo "主机初始化完成"
```

2.将除ops外的所有主机的DNS配置全都指向为proxy-master（使全部的机器访问外网就只有通过11端口，更好控制）

```shell
[root@ops ansible]# vim dns.sh
#!/bin/bash

#获取网卡名称
networkName=$(nmcli | grep "已连接" | awk '{print $1}' | sed 's/://g')

#当网卡配置文件中有DNS配置时就修改DNS地址，如果没有就添加
if (cat /etc/sysconfig/network-scripts/ifcfg-${networkName} | grep DNS); then
    sed -i 's/^DNS1.*/DNS1=192.168.100.11/g' /etc/sysconfig/network-scripts/ifcfg-${networkName}
    sed -i 's/^DNS2.*//g' /etc/sysconfig/network-scripts/ifcfg-${networkName}
else
    echo 'DNS1=192.168.100.11' >> /etc/sysconfig/network-scripts/ifcfg-${networkName}
fi

```
